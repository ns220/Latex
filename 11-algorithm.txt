\section*{Floyd's Algorithm} 




\SetKwComment{Comment}{// }{ } 
\vspace{1cm} 
\begin{algorithm}[H] 
\caption{Floyd(W [1..n, 1..n])} 
\SetAlgoLined 
\DontPrintSemicolon 
\KwIn{The weight matrix W of a graph having vertices [1..n]} 
\KwOut{The distance matrix D of the shortest paths' lengths between every pair of vertices 
[1..n]} 
$D$ $\gets$ $W$ \Comment*[r]{initially copy the weight matrix into distance matrix} 
\For{$k \gets 1$ to $n$}{ 
\For{$i \gets 1$ to $n$}{ 
\For{$j \gets 1$ to $n$}{ 
$D[i,j]$ $\gets min\{D[i,j], D[i,k] + D[k,j]\}$ 
} 
} 
} 
\Return{$D$}\; 
\end{algorithm}
